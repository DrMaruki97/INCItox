 <div class="xrm-editable-html xrm-attribute"><div class="xrm-attribute-value"><p>Select a letter of the alphabet to browse ingredients by name or enter an ingredient name in the search field below.<br>
<br>
Ingredients of interest may be found on the label of a personal care product and searched using this database. Ingredients are often referred to on the label of a personal care product by the ingredient name as it appears in the International Nomenclature of Cosmetic Ingredients (INCI), known as the INCI Name. CIR reviews ingredient safety, not products.&nbsp;<br>
<br>
Ingredients may also be browsed alphabetically, by the first letter of the INCI name; names that start with a number will be found under “#.”&nbsp; Ingredients currently on the Expert Panel for Cosmetic Ingredient Safety docket (i.e., current-year priority list), and new Scientific Literature Reviews (SLRs) not yet reviewed by the Panel, may be found at <a href="https://url.us.m.mimecastprotect.com/s/zPoXCjRB83cnNZ5pOhnAUJH?domain=cir-safety.org">https://cir-safety.org/about</a>.</p>

function sendQuery (){
	var queryString = window.location.search.substring(1);
	$.getJSON( "/FetchCIRReports?" + queryString, function( data ) {
		debugger;
		records = data.results;
		var moreRecords = data.morerecords;
		var pagingCookie = data.pagingcookie;
		var page = 2;
		if (moreRecords){
			sendAnotherQuery(queryString, pagingCookie, page);
		}
		if (searchUnlocked)
			alphaSearch();
	})
	.fail(function(error) {
		debugger;
		console.log( error );
	});

	$.getJSON( "/FetchCIRReports?" + queryString + "&part2=true", function( data ) {
			records = records.concat(data.results);
			var moreRecords = data.morerecords;
			var pagingCookie = data.pagingcookie;
	
		var page = 2;
			if (moreRecords){
				sendAnotherQuery(queryString, pagingCookie, page);
			}
			if (searchUnlocked)
				alphaSearch(); // display results of first query while subsequent fetches happen in the background
		})
		.fail(function(error) {
			console.log( error );
		});
}

function sendAnotherQuery (queryString, pagingCookie, page){ // recursively send queries until we've retrieved every record.
	$.getJSON( "/FetchCIRReports?" + queryString + "&pagingcookie=" + pagingCookie + "&page=" + page, function( data ) {
		console.log("Retrieving page " + page);
		records = records.concat(data.results);
		moreRecords = data.morerecords;
		pagingCookie = data.pagingcookie;
		page++;
		if (moreRecords){
			sendAnotherQuery(queryString, pagingCookie, page);
		}
		if (searchUnlocked)
			alphaSearch();
	})
	.fail(function(error) {
		console.log( error );
	});
}

function alphaSearch(searchString){
	searchUnlocked = true;
	if (!searchString){
		searchString = previousSearchString;
	}
	else{
		previousSearchString = searchString;
	}
	
	if (searchString == "$*"){
		searchString = ""; // handle search reset
	}
	
	let data = records;
	
	
	var searchAgainstFirstColumn = true;
	var searchAgainstSecondColumn = true;
	
	if(searchString) {
		searchString = searchString.toLowerCase();
		console.log(searchString);
		if (searchString == '#'){
			data = $.map(records,function( doc, index) {
				if(doc.pcpc_details_name && /^\d/.test(doc.pcpc_details_name) && searchAgainstFirstColumn)
				return doc;
				if(doc.pcpc_ingredientname && /^\d/.test(doc.pcpc_ingredientname) && searchAgainstSecondColumn)
				return doc;
				if(doc.pcpc_ciringredientname && /^\d/.test(doc.pcpc_ciringredientname))
				return doc;
			});
		}
		else if (searchString == '!'){
			data = $.map(records,function( doc, index) {
				if(doc.pcpc_details_name && !(/^[a-zA-Z0-9]/.test(doc.pcpc_details_name)) && searchAgainstFirstColumn)
				return doc;
				if(doc.pcpc_ingredientname && !(/^[a-zA-Z0-9]/.test(doc.pcpc_ingredientname)) && searchAgainstSecondColumn)
				return doc;
				if(doc.pcpc_ciringredientname && !(/^[a-zA-Z0-9]/.test(doc.pcpc_ciringredientname)))
				return doc;
			});
		}
		else if (searchString.startsWith('$') && searchString.length == 2){
			data = $.map(records,function( doc, index) {
				if((doc.pcpc_details_name && doc.pcpc_details_name.toLowerCase().startsWith(searchString[1]) && searchAgainstFirstColumn))
				return doc;
				if((doc.pcpc_ingredientname && doc.pcpc_ingredientname.toLowerCase().startsWith(searchString[1]) && searchAgainstSecondColumn))
				return doc;
				if((doc.pcpc_ciringredientname && doc.pcpc_ciringredientname.toLowerCase().startsWith(searchString[1])))
				return doc;
			});
		}
		else{
			data = $.map(records,function( doc, index) {
				if((doc.pcpc_details_name && doc.pcpc_details_name.toLowerCase().includes(searchString) && searchAgainstFirstColumn))
				return doc;
				if((doc.pcpc_ingredientname && doc.pcpc_ingredientname.toLowerCase().includes(searchString) && searchAgainstSecondColumn))
				return doc;
				if((doc.pcpc_ciringredientname && doc.pcpc_ciringredientname.toLowerCase().includes(searchString)))
				return doc;
				if((doc.pcpc_cirreportname && doc.pcpc_cirreportname.toLowerCase().includes(searchString)))
				return doc;
			});
		}
	}
	
	console.log(data);
	var items = [];
	
	items.push("<table class='table'><tr><td>Ingredient Name as Used:</td><td>INCI Name:</td></tr>");
	
	var previousName = "";
	var previousId = "";
	var list = null;
	$.each( data, function( index, doc ) {
		if ($("#omitNoRelatedIngredient").prop('checked') && !doc.pcpc_ingredientid){
			return;
		}
		//debugger;
		if (doc.pcpc_ingredientname == previousName && doc.pcpc_ciringredientid == previousId)
		return;
		items.push("<tr>");
		items.push("<td><a href='/cir-ingredient-status-report?id=" + doc.pcpc_ingredientid + "' target='_blank'>" + doc.pcpc_ingredientname + "</a></td>");
		if (doc.pcpc_ingredientid)
		items.push("<td>" + doc.pcpc_ciringredientname + "</td>");
		else
		items.push("<td></td>");
		items.push("</tr>");
		previousName = doc.pcpc_ingredientname;
		previousId = doc.pcpc_ciringredientid;
		
		
		console.log(doc);
	});

	items.push("</table>")
	
	console.log(items);
	$( "#documentsList").html(items.join( "" ));
